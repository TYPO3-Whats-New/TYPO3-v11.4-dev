% Feature | 92460 | Split default from all languages in page module

\begin{frame}[fragile]
	\frametitle{Backend User Interface}
	\framesubtitle{Sprachen/Übersetzungen}

	Die Ansicht "Sprachen" unter \textbf{Web -> Seite} ist jetzt konsistenter mit
	der Spaltenansicht. Die Standardsprache zeigt nur Inhaltselemente dieser
	Sprache. Backend-Benutzer können das Element "Alle Sprachen" verwenden, um zur
	Spaltenansicht wechseln, um die verfügbaren Sprachen anzuzeigen.

	\begin{figure}
		\includegraphics[width=0.9\linewidth]{BackendUserInterface/1629474065-DefaultLanguageInPageModule-2.png}
	\end{figure}

\end{frame}

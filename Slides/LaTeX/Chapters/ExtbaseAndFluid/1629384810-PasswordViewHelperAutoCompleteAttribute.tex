% Feature | 94653 | Autocomplete attribute for PasswordViewHelper

\begin{frame}[fragile]
	\frametitle{Extbase und Fluid}
	\framesubtitle{Fluid ViewHelpers}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{7}{9}\ttfamily}

	\begin{itemize}
		\item Der Fluid ViewHelper "\texttt{f:form.password}" unterstützt nun auch
			das Attribut "\texttt{autocomplete}"
		\item Dieses Attribut kan z.B. verwendet werden, um Passwort-Managern mitzuteilen,
			wie sie das Passwort-Eingabefeld behandeln sollen
		\item Zum Beispiel:
\begin{lstlisting}
<f:form.password name="newPassword" value="" autocomplete="new-password" />
\end{lstlisting}

		\item Typische Werte sind:\newline
			\small"\texttt{on}", "\texttt{off}", "\texttt{current-password}", "\texttt{new-password}"\normalsize
		\item Siehe
			\href{https://developer.mozilla.org/en-US/docs/Web/HTML/Element/input/password#allowing\_autocomplete}{MDN-Spezifikation}
			für weitere Details über mögliche Werte

	\end{itemize}

\end{frame}

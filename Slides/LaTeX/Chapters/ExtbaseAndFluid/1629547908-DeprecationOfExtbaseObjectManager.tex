% Deprecation | 94619 | Extbase ObjectManager

\begin{frame}[fragile]
	\frametitle{Extbase und Fluid}
	\framesubtitle{Extbase ObjectManager}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Der TYPO3 v11 Core verwendet den Extbase ObjectManager nicht mehr
		\item Alle zugehörigen Komponenten wurden als \textbf{veraltet} markiert
		\item Zum Beispiel schreibt der Aufruf von \texttt{ObjectManager->get()} nun
			eine veraltete Warnung in das Log
		\item Entwicklern wird empfohlen, seit TYPO3 v10 die
			\href{https://docs.typo3.org/m/typo3/reference-coreapi/master/en-us/ApiOverview/DependencyInjection/Index.html}{DependencyInjection}
			-Komponente von Symfony zu verwenden
		\item Der ObjectManager und alle zugehörigen Komponenten werden in TYPO3 v12 entfernt
		\item Siehe
			\href{https://docs.typo3.org/c/typo3/cms-core/master/en-us/Changelog/10.4/Deprecation-90803-DeprecationOfObjectManagergetInExtbaseContext.html#changelog-deprecation-90803-objectmanagerget}{Changelog}
			für Migrationsoptionen
	\end{itemize}

\end{frame}

% Feature | 90347 | Enable recursive transformation of properties in JsonView

\begin{frame}[fragile]
	\frametitle{Extbase und Fluid}
	\framesubtitle{Rekursive Eigenschaftsumwandlung}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Entwickler können jetzt die Eigenschaft "\texttt{\_recursive}" in der
			JsonView Konfiguration verwenden, um rekursive Eigenschaften aufzulösen
		\item Dies vereinfacht die Implementierung von z.B. Verzeichnissen die
			Verzeichnisse enthalten oder Kommentare, die Kommentare als Antworten enthalten, usw.
		\item Beispiele:
	\end{itemize}

	\begin{columns}[T]
		\begin{column}{.03\textwidth}
		\end{column}
		\begin{column}{.485\textwidth}
			\begin{itemize}\small
				\item 1:n relations
\begin{lstlisting}
$configuration = [
  'directories' => [
    '_descendAll' => [
      '_recursive' => ['directories']
    ]
  ]
];
\end{lstlisting}
			\end{itemize}\normalsize
		\end{column}
		\begin{column}{.485\textwidth}
			\begin{itemize}\small
				\item 1:1 relations
\begin{lstlisting}
$configuration = [
  'comment' => [
    '_recursive' => ['comment']
  ]
];
\end{lstlisting}
			\end{itemize}\normalsize
		\end{column}
	\end{columns}

\end{frame}

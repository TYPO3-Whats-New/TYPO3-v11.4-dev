% Feature | 93197 | Resolve collection types of non-persistent objects

\begin{frame}[fragile]
	\frametitle{Extbase und Fluid}
	\framesubtitle{Sammlungstypen von nicht-persistenten Objekten}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{7}{9}\ttfamily}

	\begin{itemize}
		\item Sammlungstypen werden verwendet, um eine bestimmte Klasse zu definieren,
			die in der Extbase-Klasse ObjectStorage enthalten sein soll, z.B.:
\begin{lstlisting}
/**
 * @param ObjectStorage<Item> $items
 */
public function setItems(ObjectStorage $items): void
{
  $this->items = $items;
}
\end{lstlisting}

		\item Diese Docblocks werden analysiert, damit der PropertyMapper weiß,
			wie er eingehende Anfragen zuordnen soll
		\item Entwickler können nun Auflistungstypen in Docblock-Annotationen
			auch für nicht-persistente Objekte verwenden
	\end{itemize}

\end{frame}

% Feature | 90336 | CKEditor Autolinking uses https by default

\begin{frame}[fragile]
	\frametitle{Sonstiges}
	\framesubtitle{CKEditor-Plugin "Autolinking"}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item URLs wie \texttt{example.com}, die im RTE eingegeben werden, werden
			automatisch in einen externen Link umgewandelt
		\item Dies geschieht durch das CKEditor-Plugin
			"\href{https://ckeditor.com/docs/ckeditor5/latest/features/link.html#autolink-feature}{AutoLink}"
		\item Seit TYPO3 v11.4 ist das Standardschema für diese Links: \textbf{https}
		\item Redakteure können das Schema bei Bedarf natürlich manuell auf http aktualisieren
	\end{itemize}

\end{frame}

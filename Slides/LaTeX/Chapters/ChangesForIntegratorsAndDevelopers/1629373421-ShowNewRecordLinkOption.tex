% Feature | 94765 | Introduce showNewRecordLink option

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Neue Option: \texttt{showNewRecordLink}}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Entwickler können nun den "New record"-Link von TCA-Inline-
			Elemente unabhängig von der "+" -Schaltfläche und allen anderen "level links" deaktivieren
		\item Hierfür wurde eine neue Option \texttt{showNewRecordLink} eingeführt
		\item Zum Beispiel (um die Schaltfläche "New record" zu deaktivieren):
\begin{lstlisting}
'inlineField' => [
  'label' => 'Inline without New record link',
  'config' => [
    'type' => 'inline',
    'appearance' => [
      'showNewRecordLink' => false
    ]
  ]
]
\end{lstlisting}

		\item Die folgende Option verwaltet nur noch die Anzeige der "+" -Schaltfläche:
\begin{lstlisting}
['config']['appearance']['enabledControls']['new']
\end{lstlisting}

	\end{itemize}

\end{frame}

% Feature | 94622 | New TCA type "category"

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Neuer TCA-Typ: \texttt{Kategorie} (1)}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{7}{9}\ttfamily}

	\begin{itemize}
		\item Ein neuer TCA-Feldtyp "\texttt{category}" wurde zum TYPO3 Core hinzugefügt
		\item Dies ersetzt die \texttt{CategoryRegistry} und die Methode\newline
			\small
				\texttt{ExtensionManagementUtility->makeCategorizable()}\newline
				(die wir in Zukunft als veraltet kennzeichnen werden)
			\normalsize
		\item TYPO3 generiert automatisch die TCA-Konfiguration und fügt auch das
			Datenbankfeld hinzu
		\item Beispiel für die Verwendung:
\begin{lstlisting}
$GLOBALS['TCA'][$myTable]['columns']['categories'] = [
  'config' => [
    'type' => 'category'
  ]
];

ExtensionManagementUtility::addToAllTCAtypes($myTable, 'categories');
\end{lstlisting}

	\end{itemize}

\end{frame}

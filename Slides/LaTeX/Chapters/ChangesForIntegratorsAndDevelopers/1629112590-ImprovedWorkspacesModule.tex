% Feature | 94819 | Improved Workspaces module

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Das Modul Arbeitsbereiche}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Die Benutzeroberfläche des Arbeitsbereiche-Moduls wurde verbessert\newline
			\small(z.B. kann ein Dropdown-Menü verwendet werden, um zwischen Arbeitsbereichen zu wechseln)\normalsize
		\item Administratoren können nun die Einstellungen der Arbeitsbereiche direkt bearbeiten\newline
			\small(der Zugang über das Listenmodul ist nicht mehr erforderlich)\normalsize
	\end{itemize}

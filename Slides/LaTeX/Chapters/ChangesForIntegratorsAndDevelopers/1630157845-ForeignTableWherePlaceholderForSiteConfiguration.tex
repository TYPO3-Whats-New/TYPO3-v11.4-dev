% Feature | 94662 | Add placeholder for site configuration in foreign_table_where

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{TCA: \texttt{foreign\_table\_where}}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{7}{9}\ttfamily}

	\begin{itemize}
		\item Die Einstellung \texttt{foreign\_table\_where} in der TCA unterstützt
			jetzt auch die folgenden Markierungen:

			\begin{itemize}
				\item \texttt{\#\#\#SITE:<KEY>\#\#\#}\newline
					(z.B. \texttt{\#\#\#SITE:rootPageId\#\#\#})
				\item \texttt{\#\#\#SITE:<KEY>.<SUBKEY>\#\#\#}\newline
					(z.B. \texttt{\#\#\#SITE:mySetting.categoryPid\#\#\#})
			\end{itemize}

		\item Dies ermöglicht Entwicklern den Zugriff auf die Einstellungen in der Website-Konfiguration

	\end{itemize}
\end{frame}

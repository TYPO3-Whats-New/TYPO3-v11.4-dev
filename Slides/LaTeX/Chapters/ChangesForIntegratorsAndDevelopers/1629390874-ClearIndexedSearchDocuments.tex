% Feature | 94577 | Clear indexed_search documents when content is changed

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Indizierte Suche}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item Die Systemerweiterung \texttt{EXT:indexed\_search} verfügt nun über
			eine neue Konfigurationsoption:
			\small"\texttt{basic.deleteFromIndexAfterEditing}"\normalsize
		\item Wenn diese Funktion aktiviert ist (was die Standardeinstellung ist) und eine Seite
			oder ihr Inhalt aktualisiert wird, werden die Seite und ihr Inhalt aus dem Suchindex entfernt
		\item Wenn diese Option deaktiviert ist, müssen Backend-Benutzer den
			Index nach Bedarf manuell löschen
	\end{itemize}

\end{frame}

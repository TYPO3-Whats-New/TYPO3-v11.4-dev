% Feature | 94622 | New TCA type "category"

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Neuer TCA-Typ: \texttt{Kategorie} (2)}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Entwickler können Beziehungen definieren, z.B.:
\begin{lstlisting}
$GLOBALS['TCA'][$myTable]['columns']['mainCategory'] = [
  'config' => [
    'type' => 'category',
    'relationship' => 'oneToOne'
  ]
];
\end{lstlisting}

		\item Mögliche Werte sind:
			\begin{itemize}
				\item \texttt{manyToMany} (Standard)
				\item \texttt{oneToMany}
				\item \texttt{oneToOne}
			\end{itemize}

		\item Siehe \href{https://docs.typo3.org/c/typo3/cms-core/master/en-us/Changelog/11.4/Feature-94622-NewTCATypeCategory.html}{Changelog} für weitere Details

	\end{itemize}

\end{frame}

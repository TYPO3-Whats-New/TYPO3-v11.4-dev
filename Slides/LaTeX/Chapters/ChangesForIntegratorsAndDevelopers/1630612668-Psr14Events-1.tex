\begin{frame}[fragile]
	\frametitle{Changements pour les intégrateurs et développeurs}
	\framesubtitle{Événements PSR-14 (1)}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{9}{11}\ttfamily}

	\begin{itemize}
		\item Ces événements PSR-14 sont ajoutés au noyau de TYPO3~:
			\begin{itemize}
				\item \texttt{ProcessFileListActionsEvent}\newline
					(événement lancé après que les actions du module Fichiers soient générées)
				\vspace{0.1cm}
				\item \texttt{BootCompletedEvent}\newline
					(événement lancé à chaque requête lorsque TYPO3 est entièrement chargé)
				\vspace{0.1cm}
				\item \texttt{ModifyFileDumpEvent}\newline
					(événement lancé avant le déchargement d'un fichier)
				\vspace{0.1cm}
				\item \texttt{ModifyClearCacheActionsEvent}\newline
					(événement permettant de modifier les éléments de «~cache~» dans la barre d'outil backend)
				\vspace{0.1cm}
				\item \texttt{AfterFileCommandProcessedEvent}\newline
					(événement permettant d'exécuter des tâches supplémentaires, après l'exécution d'une opération sur un fichier)
			\end{itemize}
	\end{itemize}
\end{frame}

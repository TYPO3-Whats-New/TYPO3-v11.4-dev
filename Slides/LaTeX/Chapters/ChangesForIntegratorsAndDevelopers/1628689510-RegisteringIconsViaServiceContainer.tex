% Feature | 94692 | Registering Icons via Service Container

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Benutzerdefinierte Icon-Registrierung}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Entwickler von Erweiterungen können jetzt benutzerdefinierte Symbole über einen Service-Container registrieren
		\item Der neue Ansatz verbessert die Leistung von Anfragen
		\item Speichern Sie die Konfiguration in der Datei \texttt{Configuration/Icons.php}:
\begin{lstlisting}
<?php
return [
  'myicon' => [
    'provider' => \TYPO3\CMS\Core\Imaging\IconProvider\SvgIconProvider::class,
    'source' => 'EXT:my_extension/Resources/Public/Icons/myicon.svg'
  ]
];
\end{lstlisting}

	\end{itemize}

\end{frame}

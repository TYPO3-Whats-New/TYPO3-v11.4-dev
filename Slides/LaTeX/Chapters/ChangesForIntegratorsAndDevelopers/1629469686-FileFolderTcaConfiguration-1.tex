% Feature | 94406 | Override fileFolder TCA configuration with TSconfig

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{TCA-Konguration: \texttt{fileFolder}}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{5}{6}\ttfamily}

	\begin{itemize}
		\item Die TCA-Konfiguration "\texttt{fileFolder}" wurde gestrafft
		\item Optionen sind jetzt als "\texttt{fileFolderConfig}" verfügbar:
	\end{itemize}

	\begin{columns}[T]
		\begin{column}{.04\textwidth}
		\end{column}
		\begin{column}{.48\textwidth}
			\smaller TYPO3 < v11.4 \normalsize
\begin{lstlisting}
'fieldname' => [
  'config' => [
    'type' => 'select',
    'renderType' => 'selectSingle',
    'fileFolder' => 'EXT:myext/Resources/Public/Icons',
    'fileFolder_extList' => 'svg',
    'fileFolder_recursions' => 1
  ]
]
\end{lstlisting}
		\end{column}
		\begin{column}{.48\textwidth}
			\smaller TYPO3 >= v11.4 \normalsize
\begin{lstlisting}
'fieldname' => [
  'config' => [
    'type' => 'select',
    'renderType' => 'selectSingle',
    'fileFolderConfig' => [
      'folder' => 'EXT:myext/Resources/Public/Icons',
      'allowedExtensions' => 'svg',
      'depth' => 1
    ]
  ]
]
\end{lstlisting}
		\end{column}
	\end{columns}

	\begin{itemize}
		\item Note the sub-array "\texttt{fileFolderConfig}" and the new option
			names (e.g. "\texttt{depth}") in TYPO3 >= v11.4
	\end{itemize}

\end{frame}

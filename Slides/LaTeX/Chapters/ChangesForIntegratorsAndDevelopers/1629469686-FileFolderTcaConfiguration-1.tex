% Feature | 94406 | Override fileFolder TCA configuration with TSconfig

\begin{frame}[fragile]
	\frametitle{Changes for Integrators and Developers}
	\framesubtitle{TCA Configuration: \texttt{fileFolder}}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{5}{6}\ttfamily}

	\begin{itemize}
		\item The TCA configuration "\texttt{fileFolder}" has been streamlined
		\item Options are now available as "\texttt{fileFolderConfig}":
	\end{itemize}

	\begin{columns}[T]
		\begin{column}{.01\textwidth}
		\end{column}
		\begin{column}{.495\textwidth}
			\smaller TYPO3 < v11.4 \normalsize
\begin{lstlisting}
'fieldname' => [
  'config' => [
    'type' => 'select',
    'renderType' => 'selectSingle',
    'fileFolder' => 'EXT:myext/Resources/Public/Icons',
    'fileFolder_extList' => 'svg',
    'fileFolder_recursions' => 1,
  ]
]
\end{lstlisting}
		\end{column}
		\begin{column}{.495\textwidth}
			\smaller TYPO3 >= v11.4 \normalsize
\begin{lstlisting}
'fieldname' => [
  'config' => [
    'type' => 'select',
    'renderType' => 'selectSingle',
    'fileFolderConfig' => [
      'folder' => 'EXT:myext/Resources/Public/Icons',
      'allowedExtensions' => 'svg',
      'depth' => 1,
    ]
  ]
]
\end{lstlisting}
		\end{column}
	\end{columns}

	\begin{itemize}
		\item Note the new option names (e.g. "\texttt{depth}") and the sub-array
			("\texttt{fileFolderConfig}")
	\end{itemize}

\end{frame}

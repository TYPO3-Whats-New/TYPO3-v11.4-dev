% Feature | 95044 | Support autowired LoggerInterface injection

\begin{frame}[fragile]
	\frametitle{Changements pour les intégrateurs et développeurs}
	\framesubtitle{LoggerInterface Injection}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item Services are no longer required to use
\begin{lstlisting}
Psr\Log\LoggerAwareInterface
Psr\Log\LoggerAwaireTrait
\end{lstlisting}
		\item Developers can add a constructor argument
\begin{lstlisting}
Psr\Log\LoggerInterface
\end{lstlisting}
		\item For example:
\begin{lstlisting}
use Psr\Log\LoggerInterface;
class MyClass {
  private LoggerInterface $logger;
  public function __construct(LoggerInterface $logger) {
    $this->logger = $logger;
  }
}
\end{lstlisting}
	\end{itemize}

\end{frame}

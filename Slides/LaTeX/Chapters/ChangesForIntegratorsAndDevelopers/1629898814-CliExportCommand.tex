% Feature | 84718 | Add CLI export command to EXT:impexp

\begin{frame}[fragile]
	\frametitle{Sonstiges}
	\framesubtitle{TYPO3 CLI (1)}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{9}{11}\ttfamily}

	\begin{itemize}
		\item Ein neuer Befehl wurde der TYPO3-Befehlszeilenschnittstelle (CLI) hinzugefügt:
\begin{lstlisting}
./bin/typo3 impexp:export [options] [--] [<filename>]
\end{lstlisting}
		\item Der "Export" ergänzt die bestehende "Import"-Funktionalität
		\item Der Befehl exportiert den gesamten TYPO3-Seitenbaum oder Teile davon
		\item Integratoren und Administratoren können den Export durch eine Vielzahl von
			Optionen beeinflussen (z.B. Typ der Seite, Tiefe des Seitenbaums, usw.)
		\item Exporte können in andere TYPO3-Instanzen reimportiert werden oder als
			Ausgangsseitenbäume in einer Distribution
	\end{itemize}

\end{frame}

% Feature | 94590 | Allow icon identifiers in report module registration

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Icons im Modul Berichte}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Entwickler können jetzt Icon-Identifikatoren im Registrierungs-
			Array des Backend-Moduls Reports bereitstellen
		\item Zum Beispiel:
\begin{lstlisting}
$GLOBALS['TYPO3_CONF_VARS']['SC_OPTIONS']['reports']['tx_reports']['status'] = [
  'title' =>
    'LLL:EXT:reports/Resources/Private/Language/locallang_reports.xlf:status_report_title',
  'icon' =>
    'module-reports', // icon identifiers are now possible
  'description' =>
    'LLL:EXT:reports/Resources/Private/Language/locallang_reports.xlf:status_report_description',
  'report' =>
    \TYPO3\CMS\Reports\Report\Status\Status::class
];
\end{lstlisting}

	\end{itemize}

\end{frame}
